\section{基本公式}

  E = mc^2

\begin{equation}
  \int_{0}^{\pi} \sin x\, dx = 2
\end{equation}

行内公式示例：令 $\alpha,\beta \in \mathbb{R}$，若 $\alpha>0$ 且 $\beta>0$，则 $\alpha\beta>0$。

\section{定理环境}
\begin{theorem}[示例定理]
设 $A \in \mathbb{R}^{n\times n}$。若 $A$ 可逆，则 $\rank(A)=n$。
\end{theorem}

\begin{proof}
由可逆矩阵的秩等于其阶数的性质，命题成立。
\end{proof}
